namespace = credux_dt_level

country_event = {
	id = credux_dt_level.1
	
	hide_window = yes
	is_triggered_only = no
	
	mean_time_to_happen = {
		months = 1
	}
	
	trigger = {
		has_country_flag = dt_level_0
		num_ascension_perks > 0
		has_valid_civic = civic_divine_transcendence
	}
	
	immediate = {
			add_modifier = {
				modifier = dt_level_1
			}
			set_country_flag = dt_level_1
			
			remove_country_flag = dt_level_0
			remove_modifier = dt_level_0
			
	}
}
